function initializeCoBrandDrawingPath() {
  const containerSection = document.getElementById("co-brand-pantry-matrix");
  const svgVectorLinePath = document.getElementById("svg-dotted-co-brand-trail");
  if (!containerSection || !svgVectorLinePath) return;
  
  const totalStrokeLength = svgVectorLinePath.getTotalLength();
  svgVectorLinePath.style.strokeDasharray = totalStrokeLength;
  svgVectorLinePath.style.strokeDashoffset = totalStrokeLength;
  
  const pathObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        svgVectorLinePath.style.transition = "stroke-dashoffset 2.5s cubic-bezier(0.4, 0, 0.2, 1)";
        svgVectorLinePath.style.strokeDashoffset = "0";
        animateAmenityNodesProgressively();
        pathObserver.unobserve(containerSection);
      }
    });
  }, { threshold: 0.25 });
  pathObserver.observe(containerSection);
}

function animateAmenityNodesProgressively() {
  const nodeElements = document.querySelectorAll(".amenity-path-node-circle");
  nodeElements.forEach((node, index) => {
    setTimeout(() => {
      node.classList.add("node-glow-active");
      gsap.to(node, { scale: 1.15, duration: 0.3, yoyo: true, repeat: 1 });
    }, index * 400);
  });
}

function initializeFormFocusInteractions() {
  const inputElements = document.querySelectorAll(".fluid-form-input");
  inputElements.forEach(input => {
    input.addEventListener("focus", () => {
      input.closest(".input-frame-wrapper").classList.add("aura-glow-illuminated");
    });
    input.addEventListener("blur", () => {
      input.closest(".input-frame-wrapper").classList.remove("aura-glow-illuminated");
    });
  });
}
